/**
 * Middleware para manejar rutas no encontradas (404)
 * @module middlewares/manage404
 */

/**
 * Devuelve un mensaje 404 y una imagen cuando la ruta no existe.
 * @param {import('express').Request} req - Objeto de la petición
 * @param {import('express').Response} res - Objeto de la respuesta
 * @param {import('express').NextFunction} next - Función para pasar al siguiente middleware
 */
const manage404 = (req, res, next) => {
    res.status(404).json({
        msj: "404 not found",
        img: "https://seranking.com/blog/wp-content/uploads/2021/01/404_01-min.jpg"
    });
};

module.exports = manage404;
