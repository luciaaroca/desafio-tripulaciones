/**
 * Controlador de administración
 * Gestiona ventas, clientes, productos, recursos humanos y usuarios
 * @module controllers/admin
 */

const Admin = require('../models/adminModel.js');

module.exports = {

    /**
     * Obtiene todas las ventas
     * @param {import('express').Request} req
     * @param {import('express').Response} res
     * @returns {Promise<void>}
     */
    getSales: async (req, res) => {
        try {
            const sales = await Admin.getSales();
            res.json({
                success: true,
                count: sales.length,
                data: sales
            });
        } catch (err) {
            res.status(500).json({
                success: false,
                error: err.message
            });
        }
    },

    /**
     * Obtiene todos los clientes
     * @param {import('express').Request} req
     * @param {import('express').Response} res
     * @returns {Promise<void>}
     */
    getCustomers: async (req, res) => {
        try {
            const customers = await Admin.getCustomers();
            res.json({
                success: true,
                count: customers.length,
                data: customers
            });
        } catch (err) {
            res.status(500).json({
                success: false,
                error: err.message
            });
        }
    },

    /**
     * Obtiene todos los productos
     * @param {import('express').Request} req
     * @param {import('express').Response} res
     * @returns {Promise<void>}
     */
    getProducts: async (req, res) => {
        try {
            const products = await Admin.getProducts();
            res.json({
                success: true,
                count: products.length,
                data: products
            });
        } catch (err) {
            res.status(500).json({
                success: false,
                error: err.message
            });
        }
    },

    /**
     * Obtiene empleados del departamento de Recursos Humanos
     * @param {import('express').Request} req
     * @param {import('express').Response} res
     * @returns {Promise<void>}
     */
    getHr: async (req, res) => {
        try {
            const hrEmployees = await Admin.getHr();
            res.json({
                success: true,
                count: hrEmployees.length,
                data: hrEmployees
            });
        } catch (err) {
            res.status(500).json({
                success: false,
                error: err.message
            });
        }
    },

    /**
     * Obtiene todos los usuarios del sistema
     * @param {import('express').Request} req
     * @param {import('express').Response} res
     * @returns {Promise<void>}
     */
    getAllUsers: async (req, res) => {
        try {
            const users = await Admin.getAllUsers();
            res.json({
                success: true,
                count: users.length,
                data: users
            });
        } catch (err) {
            res.status(500).json({
                success: false,
                error: err.message
            });
        }
    },

    /**
     * Obtiene un usuario por su ID
     * @param {import('express').Request} req
     * @param {import('express').Response} res
     * @returns {Promise<void>}
     */
    getUserById: async (req, res) => {
        try {
            const { user_id } = req.params;
            const user = await Admin.getUserById(user_id);
            res.json({
                success: true,
                data: user
            });
        } catch (err) {
            res.status(404).json({
                success: false,
                error: err.message
            });
        }
    },

    /**
     * Crea un nuevo usuario
     * @param {import('express').Request} req
     * @param {import('express').Response} res
     * @returns {Promise<void>}
     */
    createUser: async (req, res) => {
        try {
            const result = await Admin.createUser(req.body);
            res.status(201).json(result);
        } catch (err) {
            res.status(400).json({
                success: false,
                error: err.message
            });
        }
    },

    /**
     * Actualiza un usuario por su ID
     * @param {import('express').Request} req
     * @param {import('express').Response} res
     * @returns {Promise<void>}
     */
    updateUserById: async (req, res) => {
        try {
            const { user_id } = req.params;
            const result = await Admin.updateUserById(user_id, req.body);
            res.json(result);
        } catch (err) {
            res.status(400).json({
                success: false,
                error: err.message
            });
        }
    },

    /**
     * Elimina un usuario por su ID
     * @param {import('express').Request} req
     * @param {import('express').Response} res
     * @returns {Promise<void>}
     */
    deleteUserById: async (req, res) => {
        try {
            const { user_id } = req.params;
            const result = await Admin.deleteUserById(user_id);
            res.json(result);
        } catch (err) {
            res.status(400).json({
                success: false,
                error: err.message
            });
        }
    }
};
