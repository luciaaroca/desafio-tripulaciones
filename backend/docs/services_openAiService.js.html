/**
 * Servicio de interpretación de preguntas usando OpenAI
 * Envía consultas a GPT-4o-mini y retorna la intención o datos interpretados
 * @module services/openAiService
 */

const OpenAI = require("openai");
const schemaContext = require("../utils/schemaContext");

const client = new OpenAI({
  apiKey: process.env.OPENAI_API_KEY
});

/**
 * Interpreta una pregunta enviada por el usuario
 * Envía la pregunta al modelo GPT-4o-mini y devuelve la respuesta parseada
 *
 * @param {string} question - La pregunta o mensaje del usuario a interpretar
 * @returns {Promise<Object>} Objeto con la intención y datos interpretados por OpenAI
 * @throws {Error} Lanza un error si falla la comunicación con OpenAI o el parseo de la respuesta
 */
const interpretQuestion = async (question) => {
  const response = await client.chat.completions.create({
    model: "gpt-4o-mini",
    messages: [
      { role: "system", content: schemaContext },
      { role: "user", content: question }
    ],
    temperature: 0
  });

  return JSON.parse(response.choices[0].message.content);
};

module.exports = { interpretQuestion };
